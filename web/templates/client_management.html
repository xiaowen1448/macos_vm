const scanLoading = document.getElementById('scanLoading'); // 显示加载状态 scanBtn.style.display = 'none'; scanLoading.style.display = 'block'; // 调用扫描API fetch('/api/scan_clients', { method: 'POST', headers: { 'Content-Type': 'application/json', } }) .then(response => response.json()) .then(data => { if (data.success) { updateClientTable(data.clients); scanLoading.style.display = 'none'; }); } function toggleApiTestArea() { const apiTestArea = document.getElementById('apiTestArea'); if (apiTestArea.style.display === 'none') { apiTestArea.style.display = 'block'; } else { apiTestArea.style.display = 'none'; } } function executeApiTest() { const apiInput = document.getElementById('apiInput').value.trim(); const testResult = document.getElementById('testResult');
